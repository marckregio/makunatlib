# Android Library for Templated Classes

Abstract Classes

	API.java
	AsyncRequest.java
	BaseActivity.java
	Broadcast.java

	ADD THIS TO PROJECT LEVEL BUILD.GRADLE
	allprojects {
		repositories {
			...
			maven { url 'https://jitpack.io' }
		}
	}
	
	ADD THIS TO APP LEVEL BUILD.GRADLE
	dependencies {
		compile 'com.github.marckregio:makunatlib:[version]'
	}
